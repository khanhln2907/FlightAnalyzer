        function preprocess_data(obj)
            % Super class method to convert the time stamp from us to s
            preprocess_data@Sample(obj);
            obj.data.VelTotal = (obj.data.VelNorth.^2 + obj.data.VelEast.^2 + obj.data.VelDown.^2).^(1/2);
        end
